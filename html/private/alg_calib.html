function resultado = alg_calib(previos, senhales, params) %#ok<INUSD>

    resultado = previos;
    s=params(1);
    o=params(2);
    sc = find(isnan(resultado)); %Filas aun no procesadas
    if (~isempty(sc))
        resultado(sc)=(senhales(sc)-o)/s;
    end
